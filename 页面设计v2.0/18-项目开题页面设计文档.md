## ⚡ 交互逻辑

### 📝 开题报告编辑
```javascript
/**
 * 自动保存开题报告
 */
const autoSaveReport = debounce(async (content) => {
  await saveOpeningReport({
    projectId: currentProject.value.id,
    sectionId: currentSection.value.id,
    content: content,
    timestamp: new Date()
  });
}, 2000);

/**
 * 提交开题报告
 */
const submitReport = async () => {
  // 验证报告完整性
  const validation = await validateReport(reportData.value);
  if (!validation.valid) {
    showError(validation.message);
    return;
  }
  
  const result = await submitOpeningReport({
    projectId: currentProject.value.id,
    reportData: reportData.value,
    status: 'submitted'
  });
  
  if (result.success) {
    // 创建开题会议
    await createOpeningMeeting(currentProject.value.id);
    // 发送通知
    await sendSubmissionNotifications();
  }
};
```

### 🎯 会议管理
```javascript
/**
 * 开始开题会议
 */
const startOpeningMeeting = async (meetingId) => {
  const result = await updateMeetingStatus({
    meetingId,
    status: 'in_progress',
    startTime: new Date()
  });
  
  if (result.success) {
    // 启动实时记录
    initRealTimeRecording(meetingId);
    // 通知专家
    await notifyExperts(meetingId);
  }
};

/**
 * 专家评议
 */
const submitExpertReview = async (reviewData) => {
  const result = await submitReview({
    meetingId: currentMeeting.value.id,
    expertId: userInfo.value.id,
    scores: reviewData.scores,
    comments: reviewData.comments,
    overallGrade: reviewData.overallGrade
  });
  
  if (result.success) {
    await updateReviewStatistics();
    await checkReviewCompletion();
  }
};
```

### 📊 结果统计
```javascript
/**
 * 生成开题结果
 */
const generateOpeningResult = async (meetingId) => {
  const reviews = await getExpertReviews(meetingId);
  const statistics = calculateReviewStatistics(reviews);
  
  const result = await createOpeningResult({
    meetingId,
    statistics,
    decision: statistics.averageScore >= 80 ? 'approved' : 'conditional',
    suggestions: extractSuggestions(reviews)
  });
  
  return result;
};
```
